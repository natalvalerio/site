Exemplos de Comandos SQL

Criar Banco de Dados

-- SQLite não possui um comando específico para criar um banco de dados. 
-- Ao conectar a um banco de dados, ele será criado se não existir.
-- Exemplo: 
-- sqlite3 mydatabase.db

Criar Tabela

CREATE TABLE nome_tabela (
    id INTEGER PRIMARY KEY,
    coluna1 TEXT NOT NULL,
    coluna2 INTEGER
);

Inserir Registro

INSERT INTO nome_tabela (coluna1, coluna2) VALUES ('valor1', 123);

Editar Registro

UPDATE nome_tabela SET coluna1 = 'novo_valor' WHERE id = 1;

Apagar Registro

DELETE FROM nome_tabela WHERE id = 1;

Exibir Registros

SELECT * FROM nome_tabela;

Apagar Tabela

DROP TABLE nome_tabela;
